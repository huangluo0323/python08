案例:多线程爬虫+json存储

Queue是python中的标准库，可以直接import Queue引用;队列是线程间最常用的交换数据的形式

python下多线程的思考

对于资源，加锁是个重要的环节。因为python原生的list,dict等，都是not thread safe的。而Queue，是线程安全的，因此在满足使用条件下，建议使用队列。

  1. 初始化： class Queue.Queue(maxsize) FIFO 先进先出
  2. 包中的常用方法:
    • Queue.qsize() 返回队列的大小
    • Queue.empty() 如果队列为空，返回True,反之False
    • Queue.full() 如果队列满了，返回True,反之False
    • Queue.full 与 maxsize 大小对应
    • Queue.get([block[, timeout]])获取队列，timeout等待时间
  3. 创建一个“队列”对象
    • import Queue
    • myqueue = Queue.Queue(maxsize = 10)
  4. 将一个值放入队列中
    • myqueue.put(10)
  5. 将一个值从队列中取出
    • myqueue.get()

多线程示意图

#coding=utf-8
import requests
from lxml import etree
import json
from queue import Queue
import threading


class Qiubai_spider():
    def __init__(self):

        self.headers = {
            "User-Agent":"Mozilla/5.0 (Windows NT 10.0; WOW64; Trident/7.0; rv:11.0) like Gecko"
        }
        self.url_q = Queue()
        self.html_q = Queue()
        self.content_q = Queue()

    def get_url_list(self):
        #获取所有页面连接
        url =  "http://www.qiushibaike.com/text/page/"
        #url_list = [] 可以使用列表
        for i in range(1,14):
            #使用列表的话，使用url_lsit.append(url+str(i))
            self.url_q.put(url+str(i))

    def parse_url(self):
        while self.url_q.not_empty:
            url = self.url_q.get()
            response = requests.get(url,timeout=10,headers=self.headers) #请求url

            html = etree.HTML(response.text) #返回etree之后的html
            self.html_q.put(html)
            self.url_q.task_done()

    def parse_content(self):
        while self.html_q.not_empty:
            html = self.html_q.get()
            item_temp = html.xpath('//div[@id="content-left"]/div')

            items = []
            for item in item_temp:
                #获取用户头像地址
                avatar = item.xpath("./div[1]//img/@src")[0]
                #为头像地址添加前缀
                if  not avatar.startswith("http:"):
                    avatar = "http:"+avatar

                name = item.xpath("./div[1]//h2/text()")[0] #获取用户名

                content = item.xpath(".//div[@class='content']/span")[0].xpath('string(.)') #获取内容

                star_number = item.xpath('./div[@class="stats"]//i/text()')[0]#获取点赞数

                comment_number = item.xpath("./div[@class='stats']/span[2]/a/i/text()")[0]#获取评论数

                item = {
                    'author':name,
                    'content':content,
                    'star':star_number,
                    'comment':comment_number,
                    'avatar':avatar
                }
                items.append(item)
            self.content_q.put(items)
            self.html_q.task_done()
    def save_items(self):
        while self.content_q.not_empty:
            items = self.content_q.get()
            #encoding='utf-8-sig'
            with open('qiubai.csv','a',encoding='utf-8-sig') as f:
                json.dump(item,f,ensure_ascii=False,indent=2)
                # f.write(str(item)+'\n')
            self.content_q.task_done()


    def run(self):
        '''函数的主要逻辑实现
        '''
        '''1获取url列表'''
        thread_list = [] #存储进程对象
        thread_url = threading.Thread(target=self.get_url_list)
        thread_list.append(thread_url)
        '''2.发送网络请求'''
        for i in range(5):
            thread_parse = threading.Thread(target=self.parse_url)
            thread_list.append(thread_parse)
        '''3.提取数据'''
        thread_content = threading.Thread(target=self.parse_content)
        thread_list.append(thread_content)
        '''4.保存数据'''
        thread_save = threading.Thread(target=self.save_items)
        thread_list.append(thread_save)

        for t in thread_list:
            t.setDaemon(True)  #为每个进程设置为后台进程，效果是主进程退出子进程也会退出#为了解决程序结束无法退出的问题
            t.start()
        self.url_q.join() #让主线程等待，所有的队列为空的时候才能退出
        self.html_q.join()
        self.content_q.join()

if __name__ == "__main__":
    qiubai = Qiubai_spider()
    qiubai.run()

import csv

def save_items(self):
    while self.content_q.not_empty:
        items = self.content_q.get()
        # newline=''不设置的话会有空行存在
        #encoding='utf-8-sig'
        with open('qiubai.csv','a',newline='',encoding='utf-8-sig') as f:
            fieldnames = ['author', 'content', 'star','comment','avatar'] #设置表头
            writer = csv.DictWriter(f, fieldnames=fieldnames)
            writer.writeheader() #写入表头
            for item in items:
                writer.writerow(item)
                # json.dump(item,f,ensure_ascii=False,indent=2)
                # f.write(str(item)+'\n')
                self.content_q.task_done()
